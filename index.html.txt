    // Save Link
    async function saveLink() {
      let link = document.getElementById("youtubeLink").value.trim();
      if (!link.includes("youtube.com") && !link.includes("youtu.be")) {
        alert("❌ Only YouTube links allowed!");
        return;
      }
      await db.collection("links").add({
        url: link,
        time: new Date()
      });
      document.getElementById("youtubeLink").value = "";
      loadLinks();
    }

    // Load Links
    async function loadLinks() {
      let list = document.getElementById("linksList");
      list.innerHTML = "";
      const snapshot = await db.collection("links").orderBy("time", "desc").get();
      snapshot.forEach(doc => {
        let li = document.createElement("li");
        let a = document.createElement("a");
        a.href = doc.data().url;
        a.textContent = doc.data().url;
        a.target = "_blank";
        li.appendChild(a);
        list.appendChild(li);
      });
    }

    // Load links on page open
    window.onload = loadLinks;
